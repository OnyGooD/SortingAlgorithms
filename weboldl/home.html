A határokat és a darabszámot olvassa be a program! Beolvassa a ki.txt tartalmát, és leellenőrzi, hogy megfelel-e a feltételeknek. (darabszámnak, határoknak) ez legyen python nyelven, match lehetoseg-en belül a case 3-ban

Működés

Felhasználói bemenetek

fajl_nev = 'ki.txt'
min_ertek = int(input("Add meg a minimum határt: "))
max_ertek = int(input("Add meg a maximum határt: "))
darabszam = int(input("Add meg a sorok számát, amit vársz: "))


Ellenőrzés meghívása

ellenoriz(fajl_nev, min_ertek, max_ertek, darabszam)
: def ellenoriz(nev, minimum, maximum, darabszam):
try:
with open(nev, 'r') as file:
sor = file.readline().strip() # Egy sor beolvasása a fájlból
szamok = list(map(int, sor.split(';'))) # Számok beolvasása és konvertálása

print(f"A fájlban {len(szamok)} szám található.")

# Ellenőrizzük, hogy a fájlban lévő számok száma megegyezik-e a megadott darabszámmal
match len(szamok):
case 0:
print("A fájl üres.")
case szam if szam != darabszam:
print(f"A számok száma nem felel meg a kívántnak: {darabszam} helyett {szam}.")
case _:
# Ellenőrizzük, hogy minden szám a megadott határok között van-e
megfelel = all(minimum <= szam <= maximum for szam in szamok)
if megfelel:
print(f"Minden szám {minimum} és {maximum} között van.")
else: print(f"Van olyan szám, ami nem felel meg a határoknak ({minimum}, {maximum}).")
except FileNotFoundError:
print(f"A fájl {nev} nem található.")
except ValueError:
print("Hiba történt a fájl tartalmának feldolgozása közben. Győződj meg róla, hogy a számok helyesen vannak megadva.")
